### 示例 - DropCSS
```js
const dropcss = require("dropcss");

let html = `
    <html>
        <head></head>
        <body>
            <p>Hello World!</p>
        </body>
    </html>
`;

let css = `
    .card {
      padding: 8px;
    }
    p {
      color:red;
    }

    p:hover a:first-child {
      color: red;
    }
`;

const whitelist = /#foo|\.bar/;

let dropped = new Set();

// returns { css }
let cleaned = dropcss({
  html,
  css,
  keepText: false,
  shouldDrop: sel => {
    if (whitelist.test(sel)) return false;
    else {
      dropped.add(sel);
      return true;
    }
  }
});

console.log(cleaned.css);

console.log(dropped);
```
